%Seitennummerierungen
\newcounter{savepage}
\def\frontmatter{
	\pagestyle{rest}
    \pagenumbering{Roman}
    \setcounter{page}{1}
}

\def\mainmatter{
	\pagestyle{mainmatter}
	\setcounter{savepage}{\number\value{page}}
	\newpage
	\pagenumbering{arabic}
}

\def\backmatter{
	\pagestyle{rest}
  	\newpage
	\pagenumbering{Roman}
	\setcounter{page}{\value{savepage}}
}
